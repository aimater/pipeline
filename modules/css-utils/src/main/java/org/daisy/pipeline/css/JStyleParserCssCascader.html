82  	public JStyleParserCssCascader(URIResolver uriResolver,
83  	                               CssPreProcessor preProcessor,
84  	                               XsltProcessor xsltProcessor,
85  	                               String defaultStyleSheet,
86  	                               Medium medium,
87  	                               QName attributeName,
88  	                               CSSParserFactory parserFactory,
89  	                               RuleFactory ruleFactory,
90  	                               SupportedCSS supportedCSS,
91  	                               DeclarationTransformer declarationTransformer) {
92  		this.defaultStyleSheet = defaultStyleSheet;
93  		this.medium = medium.asMediaSpec();
94  		this.attributeName = attributeName;
95  		this.parserFactory = parserFactory;
96  		this.ruleFactory = ruleFactory;
97  		this.supportedCSS = supportedCSS;
98  		this.declarationTransformer = declarationTransformer;
99  		this.xsltProcessor = xsltProcessor;
100 		NetworkProcessor defaultNetwork = new DefaultNetworkProcessor();
101 		/*
102 		 * CSSSourceReader that handles media types supported by preProcessor. Throws a
103 		 * IOException if something goes wrong when resolving the source or if the
104 		 * pre-processing fails.
105 		 */
106 		this.cssReader = new CSSSourceReader() {
107 				public boolean supportsMediaType(String mediaType, URL url) {
108 					if ("text/css".equals(mediaType))
109 						return true;
110 					else if (mediaType == null && (url == null || url.toString().endsWith(".css")))
111 						return true;
112 					else if (preProcessor == null)
113 						return false;
114 					else
115 						return preProcessor.supportsMediaType(mediaType, url);
116 				}
117 				public CSSInputStream read(CSSSource source) throws IOException {
118 					URL url = null;
119 					InputStream is;
120 					SourceMap sourceMap = null; {
121 						switch (source.type) {
122 						case INLINE:
123 						case EMBEDDED:
124 							is = new ByteArrayInputStream(((String)source.source).getBytes());
125 							break;
126 						case URL:
127 							url = (URL)source.source;
128 							logger.debug("Fetching style sheet: " + url);
129 							Source resolved; {
130 								try {
131 									resolved = uriResolver.resolve(URLs.asURI(url).toASCIIString(), ""); }
132 								catch (javax.xml.transform.TransformerException e) {
133 									throw new IOException(e); }}
134 							if (resolved != null) {
135 								url = new URL(resolved.getSystemId());
136 								logger.debug("Resolved to :" + url); }
137 							if (resolved != null && resolved instanceof StreamSource)
138 								is = ((StreamSource)resolved).getInputStream();
139 							else
140 								is = defaultNetwork.fetch(url);
141 							// skip BOM
142 							is = new BOMInputStream(is);
143 							break;
144 						default:
145 							throw new RuntimeException("coding error");
146 						}
147 					}
148 					URL base = url != null ? url : source.base;
149 					if (!supportsMediaType(source.mediaType, url))
150 						throw new IllegalArgumentException();
151 					if (!("text/css".equals(source.mediaType)
152 					      || source.mediaType == null && (url == null || url.toString().endsWith(".css")))) {
153 						// preProcessor must be non-null
154 						try {
155 							CssPreProcessor.PreProcessingResult result
156 								= preProcessor.compile(is, url != null ? url : source.base, source.encoding);
157 							is = result.stream;
158 							if (result.sourceMap != null)
159 								sourceMap = SourceMapReader.read(result.sourceMap, result.base);
160 						} catch (RuntimeException e) {
161 							throw new IOException(
162 								(source.mediaType != null ? (source.mediaType + " p") : "P")
163 								+ "re-processing failed: " + e.getMessage(), e);
164 						}
165 					}
166 					return new CSSInputStream(is, source.encoding, base, sourceMap);
167 				}
168 			};
169 	}

173 	public Runnable transform(XMLInputValue<?> source, XMLOutputValue<?> result, InputValue<?> params) throws TransformerException {
174 		if (source == null || result == null)
175 			throw new TransformerException(new IllegalArgumentException());
176 		return () -> transform(source.ensureSingleItem().mult(2), result.asXMLStreamWriter());
177 	}

179 	private void transform(Mult<? extends XMLInputValue<?>> source, BaseURIAwareXMLStreamWriter output) throws TransformerException {
180 		Node node = source.get().asNodeIterator().next();
181 		if (!(node instanceof Document))
182 			throw new TransformerException(new IllegalArgumentException());
183 		Document document = (Document)node;
184 		BaseURIAwareXMLStreamWriter writer = output;
185 		try {
186 			URI baseURI = new URI(document.getBaseURI());
187 			URL baseURL; {
188 				if (baseURI == null || baseURI.toString().equals(""))
189 					// handling the case where base URI is empty, although this should in theory never happen
190 					// FIXME: find out why it happens
191 					baseURL = null;
192 				else
193 					baseURL = URLs.asURL(baseURI);
194 			}
195 			StyleMap styleMap;
196 			synchronized(JStyleParserCssCascader.class) {
197 				// CSSParserFactory injected in CSSAssignTraversal.<init> in CSSFactory.getUsedStyles
198 				CSSFactory.registerCSSParserFactory(parserFactory);
199 				// RuleFactory injected in
200 				// - SimplePreparator.<init> in CSSParserFactory.append in CSSFactory.getUsedStyles
201 				// - CSSTreeParser.<init> in CSSParserFactory.append in CSSFactory.getUsedStyles
202 				CSSFactory.registerRuleFactory(ruleFactory);
203 				// DeclarationTransformer injected in SingleMapNodeData.<init> in CSSFactory.createNodeData in Analyzer.evaluateDOM
204 				CSSFactory.registerDeclarationTransformer(declarationTransformer);
205 				// SupportedCSS injected in
206 				// - SingleMapNodeData.<init> in CSSFactory.createNodeData in Analyzer.evaluateDOM
207 				// - Repeater.assignDefaults in DeclarationTransformer.parseDeclaration in SingleMapNodeData.push in Analyzer.evaluateDOM
208 				// - Variator.assignDefaults in DeclarationTransformer.parseDeclaration in SingleMapNodeData.push in Analyzer.evaluateDOM
209 				CSSFactory.registerSupportedCSS(supportedCSS);
210 				styleSheet = (StyleSheet)ruleFactory.createStyleSheet().unlock();
211 				if (defaultStyleSheet != null) {
212 					StringTokenizer t = new StringTokenizer(defaultStyleSheet);
213 					while (t.hasMoreTokens()) {
214 						URL u = URLs.asURL(URLs.resolve(baseURI, URLs.asURI(t.nextToken())));
215 						if (!cssReader.supportsMediaType(null, u))
216 							logger.warn("Style sheet type not supported: " + u);
217 						else
218 							styleSheet = parserFactory.append(new CSSSource(u, (Charset)null, (String)null), cssReader, styleSheet);
219 					}
220 				}
221 				styleSheet = CSSFactory.getUsedStyles(document, null, baseURL, medium, cssReader, styleSheet);
222 				XMLInputValue<?> transformed = null;
223 				for (RuleXslt r : Iterables.filter(styleSheet, RuleXslt.class)) {
224 					Map<String,String> params = new HashMap<>();
225 					for (Declaration d: r) {
226 						if (d.size() == 1) {
227 							if (d.get(0) instanceof TermIdent || d.get(0) instanceof TermString)
228 								params.put(d.getProperty(), ((Term<String>)d.get(0)).getValue());
229 							else if (d.get(0) instanceof TermInteger)
230 								params.put(d.getProperty(), ""+((TermInteger)d.get(0)).getIntValue());
231 							else
232 								logger.warn("@xslt parameter value must be a string, ident or integer, but got " + d);
233 						} else
234 							logger.warn("@xslt parameter value must consist of exactly one part, but got " + d);
235 					}
236 					transformed = xsltProcessor.transform(
237 						URLs.resolve(URLs.asURI(r.base), URLs.asURI(r.uri)),
238 						transformed != null ? transformed : source.get(),
239 						params);
240 				}
241 				if (transformed != null) {
242 					node = transformed.ensureSingleItem().asNodeIterator().next();
243 					if (!(node instanceof Document))
244 						throw new TransformerException(
245 							new RuntimeException("XsltProcessor must return a (single) document"));
246 					document = (Document)node;
247 					// We assume that base URI did not change.
248 					// We need to recompute the stylesheet because of any possible inline styles, which
249 					// are attached to an element in the original document.
250 					styleSheet = CSSFactory.getUsedStyles(document, null, baseURL, medium, cssReader, styleSheet);
251 				}
252 				styleMap = new Analyzer(styleSheet).evaluateDOM(document, medium, false);
253 			}
254 			writer.setBaseURI(baseURI);
255 			writer.writeStartDocument();
256 			traverse(document.getDocumentElement(), styleMap, writer);
257 			writer.writeEndDocument();
258 		} catch (Exception e) {
259 			throw new TransformerException(e);
260 		} finally {
261 			styleSheet = null;
262 		}
263 	}

273 	private void traverse(Node node, StyleMap styleMap, BaseURIAwareXMLStreamWriter writer) throws XMLStreamException {
274 		if (node.getNodeType() == Node.ELEMENT_NODE) {
275 			Element elem = (Element)node;
276 			writeStartElement(writer, elem);
277 			NamedNodeMap attributes = node.getAttributes();
278 			for (int i = 0; i < attributes.getLength(); i++) {
279 				Node attr = attributes.item(i);
280 				if (!(attr.getPrefix() == null && "style".equals(attr.getLocalName())))
281 					writeAttribute(writer, attr); }
282 			Map<PseudoElement,NodeData> pseudoStyles = new java.util.HashMap<>(); {
283 				for (PseudoElement pseudo : styleMap.pseudoSet(elem))
284 					pseudoStyles.put(pseudo, styleMap.get(elem, pseudo)); }
285 			String style = serializeStyle(
286 				styleMap.get(elem),
287 				pseudoStyles,
288 				elem);
289 			if (style != null)
290 				writeAttribute(writer, attributeName, style);
291 			for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling())
292 				traverse(child, styleMap, writer);
293 			writer.writeEndElement(); }
294 		else if (node.getNodeType() == Node.COMMENT_NODE)
295 			writeComment(writer, node);
296 		else if (node.getNodeType() == Node.TEXT_NODE)
297 			writeCharacters(writer, node);
298 		else if (node.getNodeType() == Node.PROCESSING_INSTRUCTION_NODE)
299 			writeProcessingInstruction(writer, node);
300 		else
301 			throw new UnsupportedOperationException("Unexpected node type");
302 	}
