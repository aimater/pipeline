1   package org.daisy.pipeline.braille.css;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   import java.net.URI;
6   import java.net.URL;
7   
8   import static com.google.common.collect.Iterables.filter;
9   
10  import cz.vutbr.web.css.Declaration;
11  import cz.vutbr.web.css.TermIdent;
12  import cz.vutbr.web.css.TermInteger;
13  import cz.vutbr.web.css.TermURI;
14  
15  import org.daisy.braille.css.InlineStyle;
16  import org.daisy.braille.css.RuleTextTransform;
17  import org.daisy.common.file.URLs;
18  
19  import org.daisy.pipeline.braille.common.Query;
20  import org.daisy.pipeline.braille.common.Query.MutableQuery;
21  import static org.daisy.pipeline.braille.common.Query.util.mutableQuery;
22  
23  public class TextTransformParser {
24  
25  	/**
26  	 * Parse <code>@text-transform</code> rules with <code>system: braille-translator</code> and
27  	 * return them as a set of named queries.
28  	 */
29  	public static Map<String,Query> getBrailleTranslatorQueries(String style, URI baseURI, Query baseQuery) {
30  		if (style != null && !"".equals(style)) {
31  			Map<String,Query> queries = null;
32  			for (RuleTextTransform rule : filter(new InlineStyle(style), RuleTextTransform.class))
33  				for (Declaration d : rule)
34  					if (d.getProperty().equals("system")
35  					    && d.size() == 1
36  					    && d.get(0) instanceof TermIdent
37  					    && "braille-translator".equals(((TermIdent)d.get(0)).getValue())) {
38  						MutableQuery query = mutableQuery(baseQuery);
39  						for (Declaration dd : rule)
40  							if (dd.getProperty().equals("system")
41  							    && dd.size() == 1
42  							    && dd.get(0) instanceof TermIdent
43  							    && "braille-translator".equals(((TermIdent)dd.get(0)).getValue()))
44  								;
45  							else if (!dd.getProperty().equals("system")
46  							         && dd.size() == 1
47  							         && (dd.get(0) instanceof TermIdent
48  							             || dd.get(0) instanceof TermURI
49  							             || dd.get(0) instanceof TermInteger)) {
50  								String key = dd.getProperty();
51  								String value;
52  								if (dd.get(0) instanceof TermURI) {
53  									URL cssBase = ((TermURI)dd.get(0)).getBase(); // this is always null because the source is a string
54  									value = URLs.resolve(cssBase != null ? URLs.asURI(cssBase) : baseURI,
55  									                     URLs.asURI(((TermURI)dd.get(0)).getValue()))
56  										.toASCIIString();
57  								} else {
58  									if (dd.get(0) instanceof TermInteger)
59  										value = "" + ((TermInteger)dd.get(0)).getIntValue();
60  									else
61  										value = "" + dd.get(0).getValue();
62  									if (query.containsKey(key))
63  										query.removeAll(key);
64  								}
65  								if (key.equals("contraction") && value.equals("no"))
66  									query.removeAll("grade");
67  								// FIXME: support this in Liblouis translator
68  								if (key.equals("table") || key.equals("liblouis-table")) {
69  									query.removeAll("locale");
70  									query.removeAll("type");
71  									query.removeAll("contraction");
72  									query.removeAll("grade");
73  									query.removeAll("dots");
74  									query.removeAll("direction");
75  								}
76  								query.add(key, value);
77  							} else {
78  								query = null;
79  								break;
80  							}
81  						if (query != null) {
82  							if (queries == null) queries = new HashMap<>();
83  							String name = rule.getName();
84  							queries.put(name == null ? "auto" : name, query);
85  						}
86  						break;
87  					}
88  			return queries;
89  		} else
90  			return null;
91  	}
92  }
