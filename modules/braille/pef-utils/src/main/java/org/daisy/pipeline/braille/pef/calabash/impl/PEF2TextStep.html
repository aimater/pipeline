100 	@Override
101 	public void run() throws SaxonApiException {
102 		super.run();
103 		MutableQuery q = mutableQuery(query(getOption(_file_format, "")));
104 		addOption(_line_breaks, q);
105 		addOption(_page_breaks, q);
106 		addOption(_pad, q);
107 		addOption(_charset, q);
108 		RuntimeValue tableQuery = getOption(_table);
109 		if (tableQuery != null) {
110 			Table table;
111 			try {
112 				logger.debug("Finding table for query: " + tableQuery);
113 				table = tableProvider.get(query(tableQuery.getString())).iterator().next();
114 				logger.debug("Found table: " + table); }
115 			catch (NoSuchElementException e) {
116 				
117 				// this fallback is done because in dtbook-to-pef we use the
118 				// query (locale:...) which does not always match something
119 				// FIXME: https://github.com/daisy/pipeline-mod-braille/issues/75
120 				logger.warn("Table not found, falling back to en-US table.");
121 				table = tableProvider.get(EN_US).iterator().next(); }
122 			q.add("table", table.getIdentifier()); }
123 		logger.debug("Finding file format for query: " + q);
124 		Iterable<FileFormat> fileFormats = fileFormatProvider.get(q);
125 		if (!fileFormats.iterator().hasNext()) {
126 			throw new XProcException(step, "No file format found for query: " + q); }
127 		for (FileFormat fileFormat : fileFormats) {
128 			try {
129 				logger.debug("Storing PEF to file format: " + fileFormat);
130 				
131 				// Initialize output directory
132 				File textDir = new File(new URI(getOption(_dir_href).getString()));
133 				textDir.mkdirs();
134 				
135 				// Read source PEF
136 				ByteArrayOutputStream s = new ByteArrayOutputStream();
137 				Serializer serializer = runtime.getProcessor().newSerializer();
138 				serializer.setOutputStream(s);
139 				serializer.setCloseOnCompletion(true);
140 				serializer.serializeNode(source.read());
141 				serializer.close();
142 				InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
143 				s.close();
144 				
145 				// Parse pattern
146 				String singleVolumeName = getOption(_single_volume_name, "");
147 				String pattern = getOption(_name_pattern, "");
148 				if (pattern.isEmpty())
149 					pattern = "volume-{}";
150 				int match = pattern.indexOf("{}");
151 				if (match < 0 || match != pattern.lastIndexOf("{}")) {
152 					// Output to single file
153 					convertPEF2Text(pefStream,
154 							new File(textDir, singleVolumeName + fileFormat.getFileExtension()), fileFormat);
155 				} else {
156 					// Split PEF
157 					pattern = pattern.replaceAll("'", "''")
158 							.replaceAll("([0#\\.,;%\u2030\u00A4-]+)", "'$1'");
159 					// Recalculate after replacement
160 					match = pattern.indexOf("{}");
161 					File splitDir = new File(textDir, "split");
162 					splitDir.mkdir();
163 					// FIXME: to validating the result PEFs, get a PEFValidator instance
164 					// (implemented in dotify.task.impl) through the streamline API.
165 					PEFFileSplitter splitter = new PEFFileSplitter(x -> true);
166 					String prefix = PEFFileSplitter.PREFIX;
167 					String postfix = PEFFileSplitter.POSTFIX;
168 					splitter.split(pefStream, splitDir, prefix, postfix);
169 					File[] pefFiles = splitDir.listFiles();
170 					String formatPattern = pattern.substring(0, match);
171 					int nWidth; {
172 						try {
173 							nWidth = Integer.parseInt(getOption(_number_width, "")); }
174 						catch (NumberFormatException e) {
175 							nWidth = 0; }}
176 					if (nWidth == 0)
177 						formatPattern += "###"; // Assume max 999 volumes
178 					else
179 						while (nWidth > 0) { formatPattern += "0"; nWidth--; }
180 					formatPattern += pattern.substring(match + 2);
181 					NumberFormat format = new DecimalFormat(formatPattern);
182 					for (File pefFile : pefFiles) {
183 						InputStream is = new FileInputStream(pefFile);
184 						if (pefFiles.length == 1 && !singleVolumeName.isEmpty()) {
185 							// Output to single file
186 							convertPEF2Text(is, new File(textDir, singleVolumeName + fileFormat.getFileExtension()), fileFormat);
187 						} else {
188 							String pefName = pefFile.getName();
189 							if (pefName.length() <= prefix.length() + postfix.length()
190 							    || !pefName.substring(0, prefix.length()).equals(prefix)
191 							    || !pefName.substring(pefName.length() - postfix.length()).equals(postfix)) {
192 								is.close();
193 								throw new RuntimeException("Coding error");
194 							}
195 							String textName = format.format(
196 									Integer.parseInt(pefName.substring(prefix.length(), pefName.length() - postfix.length())));
197 							convertPEF2Text(is,
198 									new File(textDir, textName + fileFormat.getFileExtension()),
199 									fileFormat);
200 						}
201 						is.close();
202 						if (!pefFile.delete()) pefFile.deleteOnExit();
203 					}
204 					pefStream.close();
205 					if (!splitDir.delete()) splitDir.deleteOnExit();
206 				}
207 				return; }
208 			catch (Exception e) {
209 				logger.error("Storing PEF to file format '" + fileFormat + "' failed", e); }}
210 		throw new XProcException(step, "pef:pef2text failed");
211 	}

213 	private void convertPEF2Text(InputStream pefStream, File textFile, FileFormat fileFormat)
214 			throws ParserConfigurationException, SAXException, IOException, UnsupportedWidthException {
215 		// Create EmbosserWriter
216 		OutputStream textStream = new FileOutputStream(textFile);
217 		EmbosserWriter writer = fileFormat.newEmbosserWriter(textStream);
218 		
219 		// Parse PEF to text
220 		PEFHandler.Builder builder = new PEFHandler.Builder(writer);
221 		builder.range(null).align(Alignment.LEFT).offset(0);
222 		parsePefFile(pefStream, builder.build());
223 		textStream.close();
224 	}

286 	// copied from org.daisy.braille.facade.PEFConverterFacade because it is no longer static
287 	/**
288 	 * Parses the given input stream using the supplied PEFHandler.
289 	 * @param is the input stream
290 	 * @param ph the PEFHandler
291 	 * @throws ParserConfigurationException
292 	 * @throws SAXException
293 	 * @throws IOException
294 	 * @throws UnsupportedWidthException
295 	 */
296 	private static void parsePefFile(InputStream is, PEFHandler ph)
297 			throws ParserConfigurationException, SAXException, IOException, UnsupportedWidthException {
298 		
299 		SAXParserFactory spf = SAXParserFactory.newInstance();
300 		spf.setNamespaceAware(true);
301 		SAXParser sp = spf.newSAXParser();
302 		try {
303 			sp.parse(is, ph); }
304 		catch (SAXException e) {
305 			if (ph.hasWidthError())
306 				throw new UnsupportedWidthException(e);
307 			else
308 				throw e; }
309 	}
