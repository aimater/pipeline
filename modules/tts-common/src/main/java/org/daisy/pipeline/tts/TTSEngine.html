1   package org.daisy.pipeline.tts;
2   
3   import java.util.Collection;
4   import java.util.List;
5   
6   import javax.sound.sampled.AudioFormat;
7   
8   import net.sf.saxon.s9api.XdmNode;
9   
10  import org.daisy.pipeline.audio.AudioBuffer;
11  import org.daisy.pipeline.tts.AudioBufferAllocator.MemoryException;
12  import org.daisy.pipeline.tts.TTSRegistry.TTSResource;
13  import org.daisy.pipeline.tts.TTSService.Mark;
14  import org.daisy.pipeline.tts.TTSService.SynthesisException;
15  
16  /**
17   * Classes that inherit from TTSEngine are the ones that deal with adapting
18   * external TTS processors (e.g. eSpeak and SAPI) to the Pipeline interface,
19   * which is used notably by the TextToPcm threads in the conversion from SSML to
20   * mp3. TTSEngines are meant to be allocated for every new session so that the
21   * end user can change the TTSEngine's parameters between two Pipeline jobs.
22   * Most of the methods of TTSEngine must be thread-safe.
23   */
24  public abstract class TTSEngine {
25  
26  	/**
27  	 * @param provider is the service from which the TTSEngine has been
28  	 *            allocated.
29  	 */
30  	protected TTSEngine(TTSService provider) {
31  		mProvider = provider;
32  	}
33  
34  	/**
35  	 * @return the service from which the TTSEngine has been allocated.
36  	 */
37  	public TTSService getProvider() {
38  		return mProvider;
39  	}
40  
41  	protected TTSService mProvider;
42  
43  	/**
44  	 * This method must be thread-safe. But @param threadResources is here to
45  	 * prevent the service from locking internal resources.
46  	 * 
47  	 * @param sentence is the sentence to synthesize. It is the serialized
48  	 *            result of the SSML conversion performed by the XSLT whose URL
49  	 *            is returned by {@link TTSService#getSSMLxslTransformerURL()},
50  	 *            or <code>null</code> if that method returns <code>null</code>.
51  	 * @param xmlSentence is the XML version of @param sentence. It should not
52  	 *            be used unless one needs information that has been lost during
53  	 *            the serialization process (e.g. marks or CSS)
54  	 * @param voice is the voice the synthesizer must use. It is guaranteed to
55  	 *            be one of those returned by getAvailableVoices(). This
56  	 *            parameter can't be null.
57  	 * @param threadResources is the object returned by
58  	 *            allocateThreadResource(). It may contain small persistent
59  	 *            buffers, opened file streams, TCP connections and so on. The
60  	 *            boolean field 'released' is guaranteed to be false, i.e. the
61  	 *            resource provided is always valid and will remain so during
62  	 *            the call.
63  	 * @param marks are the returned pairs (markName, offset-in-output)
64  	 *            corresponding to the ssml:marks of @param sentence. The order
65  	 *            must be kept. The provided list is always empty. The
66  	 *            offset-in-outputs are relative to the new buffers returned by
67  	 *            synthesize(). That is, they start at 0. If the service doesn't
68  	 *            handle SSML marks (i.e. endingmark() returns null). This
69  	 *            parameter may be set to null.
70  	 * @param expectedMarks are the mark names extracted from @param xmlSentence. The
71  	 * 	          TTS engine can use them if it doesn't play well with complicated/long
72  	 *            mark names, but it will harder to detect errors. It does include
73  	 *            the TTS provider's ending mark. The ending-mark can be set to anything if
74  	 *            the TTS processor or @param voice cannot handle marks. 
75  	 * @param bufferAllocator is the object that the TTS Service must use to
76  	 *            allocate new audio buffers.
77  	 * @param retry is true when this is not the first time the thread attempts
78  	 *            to synthesize @param sentence. In such cases, you may
79  	 *            reinitialize the connection to remote TTS servers if there are
80  	 *            any.
81  	 * 
82  	 * 
83  	 * @return a list of adjacent PCM chunks produced by the TTS processor.
84  	 */
85  	abstract public Collection<AudioBuffer> synthesize(String sentence, XdmNode xmlSentence,
86  	        Voice voice, TTSResource threadResources, List<Mark> marks, List<String> expectedMarks,
87  	        AudioBufferAllocator bufferAllocator, boolean retry) throws SynthesisException,
88  	        InterruptedException, MemoryException;
89  
90  	/**
91  	 * @return the audio format (sample rate etc.) of the data produced by
92  	 *         synthesize(). The engine is assumed to use the same audio format
93  	 *         every time. It is okay to return null before the first call to
94  	 *         synthesize(), though it is better to return non-null values for
95  	 *         optimization purposes. It must, however, return a non-null value
96  	 *         once synthesize() has been called. Must be thread-safe.
97  	 */
98  	abstract public AudioFormat getAudioOutputFormat();
99  
100 	/**
101 	 * Need not be thread-safe. This method is called from the main thread.
102 	 */
103 	abstract public Collection<Voice> getAvailableVoices() throws SynthesisException,
104 	        InterruptedException;
105 
106 	/**
107 	 * This method must be thread-safe. It allocates new resources (such as TCP
108 	 * connections) unique for each thread. Allocations can be made on-the-fly
109 	 * from different threads. It must not catch InterruptuedExceptions.
110 	 * 
111 	 * @return the resources. Must not be null.
112 	 * @throws SynthesisException
113 	 */
114 	public TTSResource allocateThreadResources() throws SynthesisException,
115 	        InterruptedException {
116 		return new TTSResource();
117 	}
118 
119 	/**
120 	 * This method must be thread-safe. Deallocations may be performed from
121 	 * different threads but are always performed in the same thread as the one
122 	 * exploiting @param resources.
123 	 * 
124 	 * @param resources is the object returned by allocateThreadResource()
125 	 */
126 	public void releaseThreadResources(TTSResource resources) throws SynthesisException,
127 	        InterruptedException {
128 	}
129 
130 	/**
131 	 * Force interruption of the execution of synthesize() when the thread-level
132 	 * interruption is not enough to make synthesize() finish. Must be
133 	 * thread-safe, although the method must not wait for locks.
134 	 * 
135 	 * @param resource is the same as the one provided to synthesize()
136 	 */
137 	public void interruptCurrentWork(TTSResource resource) {
138 	}
139 
140 	/**
141 	 * Need not be thread-safe. This method is called from the main thread.
142 	 */
143 	public int getOverallPriority() {
144 		return 1;
145 	}
146 
147 	/**
148 	 * @return the number of text-to-pcm threads reserved for this TTS
149 	 *         processor. Different values than zero make only sense if the TTS
150 	 *         speed is limited by the number of opened resources rather than by
151 	 *         the number of cores. In such cases, we want to maximize the time
152 	 *         spent on using the TTS resources by avoiding using threads for
153 	 *         something else (e.g. audio encoding or calling other TTS
154 	 *         processors).
155 	 */
156 	public int reservedThreadNum() {
157 		return 0;
158 	}
159 
160 	/**
161 	 * @return the average number of milliseconds the TTS processors is expected
162 	 *         to spend to process a single word on a single CPU core. It
163 	 *         doesn't need to be accurate. It is used for tuning the timeouts.
164 	 */
165 	public int expectedMillisecPerWord() {
166 		return 100;
167 	}
168 
169 	/**
170 	 * @return the name of the mark that will be added to check whether all the
171 	 *         SSML have been successfully synthesized. TTS processors that
172 	 *         cannot handle marks must return null. Must be thread-safe.
173 	 */
174 	public String endingMark() {
175 		return null; //marks not handled
176 	}
177 }
